\question For $f(x)=2e^x$ and $g(x)=x^2$, find the following:
\begin{matching}
\mitem{1}{$(f\circ g)(x)$}{A}{$2e^{(x^2)}$}
\mitem{2}{$(g\circ f)(x)$}{B}{$4e^{2x}$}
\mitem{3}{$(f\cdot g)(x)$}{C}{$2x^2e^{x}$}
\mitem{4}{$(f+g)(x)$}{D}{$2e^{x}+x^2$}
\end{matching}
